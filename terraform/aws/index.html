Terraform on AWS

Provider and Backend Configuration

        
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 3.63"
    }

    cloudflare = {
      source  = "cloudflare/cloudflare"
      version = "~> 3.0"
    }
  }

  backend "remote" {
    organization = "morriscloud"

    workspaces {
      name = "static-website-playground-aws-terraform"
    }
  }
}
